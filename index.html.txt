    <script>
        let currentPricingMode = 'self';
        let selectedCourses = {};
        let selectedPaymentPlan = null;
        let selectedMentor = null;

        // Initialize date picker
        document.getElementById('session-date').min = new Date().toISOString().split('T')[0];

        // Pricing Toggle
        function togglePricing(mode) {
            currentPricingMode = mode;
            
            // Update toggle buttons
            document.querySelectorAll('.toggle-option').forEach(btn => {
                btn.classList.remove('active');
            });
            event.target.classList.add('active');
            
            // Update all course prices
            updateAllCoursePrices();
            
            // Show/hide mentor booking section
            const mentorBooking = document.getElementById('mentor-booking');
            if (mode === 'mentor') {
                mentorBooking.classList.add('show');
            } else {
                mentorBooking.classList.remove('show');
            }
        }

        function updateAllCoursePrices() {
            const courses = ['soc', 'threat', 'splunk', 'cloud', 'grc', 'ai'];
            const prices = {
                'soc': { mentor: 1800, self: 1200 },
                'threat': { mentor: 2000, self: 1400 },
                'splunk': { mentor: 1600, self: 1100 },
                'cloud': { mentor: 1900, self: 1300 },
                'grc': { mentor: 2000, self: 1400 },
                'ai': { mentor: 2100, self: 1450 }
            };
            
            courses.forEach(course => {
                const priceElement = document.getElementById(`${course}-price`);
                const price = prices[course][currentPricingMode];
                priceElement.textContent = `$${price.toLocaleString()}`;
                
                const typeElement = priceElement.nextElementSibling;
                typeElement.textContent = currentPricingMode === 'self' ? 'Self-Paced' : 'With Mentor';
            });
        }

        // Course Selection
        function selectCourse(courseId, mentorPrice, selfPrice) {
            // Remove previous selections
            document.querySelectorAll('.course-card').forEach(card => {
                card.classList.remove('selected');
            });
            
            // Add selection to current card
            event.currentTarget.classList.add('selected');
            
            // Show course preview
            const preview = event.currentTarget.querySelector('.course-preview');
            document.querySelectorAll('.course-preview').forEach(p => {
                p.style.display = 'none';
            });
            preview.style.display = 'block';
            
            // Store selection
            const currentPrice = currentPricingMode === 'mentor' ? mentorPrice : selfPrice;
            selectedCourses = {
                id: courseId,
                name: event.currentTarget.querySelector('h3').textContent,
                price: currentPrice,
                mentorPrice: mentorPrice,
                selfPrice: selfPrice,
                type: currentPricingMode === 'mentor' ? 'With Mentor' : 'Self-Paced'
            };
            
            updateCourseSummary();
        }

        function updateCourseSummary() {
            const summary = document.getElementById('course-summary');
            const courseName = document.getElementById('selected-course-name');
            const courseType = document.getElementById('selected-course-type');
            const coursePrice = document.getElementById('selected-course-price');
            
            courseName.textContent = selectedCourses.name;
            courseType.textContent = selectedCourses.type;
            coursePrice.textContent = `$${selectedCourses.price.toLocaleString()}`;
            
            // Update payment options
            updatePaymentOptions();
            
            summary.style.display = 'block';
            summary.scrollIntoView({ behavior: 'smooth' });
        }

        function updatePaymentOptions() {
            const basePrice = selectedCourses.price;
            const fullPaymentPrice = Math.round(basePrice * 0.95); // 5% discount
            const monthlyPrice = Math.round(basePrice / 3);
            const weeklyPrice = Math.round(basePrice / 12);
            
            document.getElementById('full-payment-price').textContent = `$${fullPaymentPrice.toLocaleString()}`;
            document.getElementById('monthly-payment-price').textContent = `$${monthlyPrice.toLocaleString()}/month`;
            document.getElementById('weekly-payment-price').textContent = `$${weeklyPrice.toLocaleString()}/week`;
        }

        // Animate progress bars when they come into view
        const observerCallback = (entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    const progressBars = entry.target.querySelectorAll('.progress-fill');
                    progressBars.forEach(bar => {
                        const width = bar.style.width;
                        bar.style.width = '0%';
                        setTimeout(() => {
                            bar.style.width = width;
                        }, 100);
                    });
                }
            });
        };

        const observer = new IntersectionObserver(observerCallback);
        document.querySelectorAll('.progress-bar').forEach(bar => {
            observer.observe(bar.parentElement);
        });
