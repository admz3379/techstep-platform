HTML Usage

<!-- Image -->
<img src="media/images/courses/your-image.jpg" 
     alt="Course Thumbnail" 
     class="w-full rounded-lg">

<!-- Video -->
<video controls class="w-full rounded-lg">
    <source src="media/videos/demos/your-video.mp4" 
            type="video/mp4">
</video>

<!-- Background Image -->
<div style="background-image: url('media/images/backgrounds/hero.jpg')"
     class="bg-cover bg-center h-96">
</div>
